üêç Python

Python 
import requests

# Your API key from the Developer Portal
API_KEY = "sk_beta_your_api_key_here"
BASE_URL = "https://api.getbrikk.com"

# Call an agent
response = requests.post(
    f"{BASE_URL}/api/v1/marketplace/agents/call",
    headers={
        "X-API-Key": API_KEY,
        "Content-Type": "application/json"
    },
    json={
        "agent_id": "csv-analyzer",
        "input": {
            "file_url": "https://example.com/data.csv",
            "analyze": ["trends", "outliers"]
        }
    }
)

result = response.json()
print(f"Success: {result['success']}")
print(f"Result: {result['result']}")
print(f"Request ID: {result['request_id']}")

üì¶ Node.js

Node.js 
const axios = require('axios');

// Your API key from the Developer Portal
const API_KEY = 'sk_beta_your_api_key_here';
const BASE_URL = 'https://api.getbrikk.com';

// Call an agent
async function callAgent() {
    try {
        const response = await axios.post(
            `${BASE_URL}/api/v1/marketplace/agents/call`,
            {
                agent_id: 'email-summarizer',
                input: {
                    emails: [
                        {
                            from: 'john@example.com',
                            subject: 'Project Update',
                            body: 'Hi team, here is the latest update...'
                        }
                    ]
                }
            },
            {
                headers: {
                    'X-API-Key': API_KEY,
                    'Content-Type': 'application/json'
                }
            }
        );

        console.log('Success:', response.data.success);
        console.log('Result:', response.data.result);
        console.log('Request ID:', response.data.request_id);
    } catch (error) {
        console.error('Error:', error.response?.data || error.message);
    }
}

callAgent();

üîë Authentication

All agent calls require authentication using your API key via the X-API-Key header.

‚ö†Ô∏è Security Best Practice: Never expose your API key in client-side code or public repositories. Always use environment variables or secure configuration management.

Getting Your API Key

  1. Open the Developer Portal using your magic link
  2. Your API key will be displayed in the "API Access" section
  3. Copy the key and store it securely
  4. Use it in the X-API-Key header for all API requests
